```shell
#!/bin/bash

echo "Shell 传递参数实例！";
echo "执行的文件名：$0";
echo "第一个参数为：$1";
echo "第二个参数为：$2";
echo "第三个参数为：$3";

# 传递到脚本的参数个数
echo "$#"

# 单字符串显示所有参数 以"$1 $2 … $n"的形式输出所有参数。
echo "$*"

# 脚本运行的当前进程id号
echo "$$"

# 后台运行的最后一个进程id号
echo "$!"

# 与$*相同，但是使用时加引号，并在引号中返回每个参数。
echo "$@"

# 显示最后命令的退出状态
echo "$?"

# $@ vs $*
echo "-- \$* 演示 ---"
for i in "$*"; do
    echo $i
done

echo "-- \$@ 演示 ---"
for i in "$@"; do
    echo $i
done
```

执行：

> ./paramShell.sh 1 2 3
